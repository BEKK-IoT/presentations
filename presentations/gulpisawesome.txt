# gulpfile.js

```javascript
var gulp = require('gulp');
var babel = require('babel');

gulp.src(['./src/**.js'])
  .pipe(babel())
  .pipe(gulp.dest('dist/'));
```

# Streams

```javascript
mergestream = require('merge-stream'),

function app() { return gulp.src(config.SRC.JS).pipe(bable()); }
function less() { return gulp.src(config.SRC.LESS).pipe(less()); }

function build() {
  return mergestream(app(), less());
}

```

---

# Streams (queue)

```javascript
streamqueue = require('streamqueue');

function app() { return gulp.src(config.SRC.JS).pipe(bable()); }
function lib() { return gulp.src(config.SRC.LIB); }

function buildJS() {
  return streamqueue({objectMode: true}, lib(), app())
          .pipe(concat('myapp.js'));   
}


```
