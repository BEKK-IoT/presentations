# Streams (queue)

```javascript
...
streamqueue = require('streamqueue');

function build() {
  let app = () => gulp.src(config.SRC.JS).pipe(bable());
  let lib = () => gulp.src(config.SRC.LIB);

  return streamqueue({objectMode: true}, lib(), app())
            .pipe(concat('myapp.js')); 
}
```
